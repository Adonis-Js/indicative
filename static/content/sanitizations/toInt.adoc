Converts value to an integer using `parseInt`.
 
[source, js]
----
const sanitizationRules = {
  age: 'to_int'
}
 
// or
const sanitizationRules = {
  age: [
    rule('to_int')
  ]
}
----
For customized build, you can import this rule as follows.
[source, js]
----
import { toInt } from 'indicative/builds/validations'
----
