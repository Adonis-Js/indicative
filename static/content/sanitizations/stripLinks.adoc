category: sanitizations
---

Strips `a` tags from a given string.

[source, js]
----
const sanitizationRules = {
  message: 'strip_links'
}

// or
const sanitizationRules = {
  message: [
    rule('strip_links')
  ]
}
----
For customized build, you can import this rule as follows.
[source, js]
----
import { stripLinks } from 'indicative/builds/sanitizations'
----
